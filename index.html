--Filter()--

let numeros = [20, 30, 40, 50, 60, 70, 80, 90, 100, 120, 130];

function checkEven(numero){

if(numero % 3 == 0)

return true;

else

return false;

}

let NumerosPares = numeros.filter(checkEven);

console.log(NumerosPares);

--find()--

let numeros = [35, 21, 34, 40, 55];

function esPar(elemento){

return elemento % 2 == 0;

}

let numeroPar = numeros.find(esPar);

console.log(numeroPar);

--FindLast()--

const numeros = [5, 14, 25, 44, 171];

const numeros = [5,14,25,44,171]
