01-Arrays functions

--concat()--

const arr1 = ["Messi", "Neymar"];

const arr2 = ["Cristiano", "Bale"];

const futbolistas = arr1.concat(arr2);

console.log(futbolistas);

--at()--

const games = ["Overwatch", "FIFA", "GTA", "Assasins creed"];

let index = 1;

let game = games.at(index);

console.log(game);

--CopyWithin()--

let words = ["Messi", "CR7", "Neymar", "Benzema", "Suarez"];

words.copyWithin(1,3,4);

console.log(words);

--Entries()--

const movies = ["Interestelar", "Armageddon", "No Time To Die"];

let iterator = movies.entries();

for(let entry of iterator){

console.log(entry);

}

--Every()--

const myArray = [20, 30, 40, 50];

const areEvery = myArray.every(x => x < 60);

console.log(areEvery);

--fill()--

var valorant = ['Jett', 'Omen', 'Fade', 'viper','Reyna'];

valorant.fill('Neon');

console.log(valorant);

--Filter()--

let numeros = [20, 30, 40, 50, 60, 70, 80, 90, 100, 120, 130];

function checkEven(numero){

if(numero % 3 == 0)

return true;

else

return false;

}

let NumerosPares = numeros.filter(checkEven);

console.log(NumerosPares);

--find()--

let numeros = [35, 21, 34, 40, 55];

function esPar(elemento){

return elemento % 2 == 0;

}

let numeroPar = numeros.find(esPar);

console.log(numeroPar);

--FindIndex()--

let Personajes = ["Ragnar Lothbrok", "Peter Parker", "Rick Grimes", "Goku", "Bruce Wayne"];

let index = Personajes.findIndex((Personaje) => Personaje === "Goku");

console.log(index);

--FindLast()--

const numeros = [5, 14, 25, 44, 171];

const encontrar = numeros.findLast((elemento) => elemento < 45);

console.log(encontrar);

--FindLastIndex()--

const numeros = [5,14,25,44,171]

const PosicionNumero = (elemento) => elemento < 45 ;

console.log(numeros.findLastIndex(PosicionNumero));
